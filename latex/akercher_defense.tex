\begin{gather*}
\label{eqn:euler1} \pd{\rho}{t} + \diverge{\left(\rho\mbf{v}\right)} = 0 \text{ ,} \\

\begin{frame}[fragile]
\frametitle{Ideal magnetohydrodynamics}
\only<1-1>{
\begin{gather*}
\pd{\rho}{t} + \diverge{\left(\rho\mbf{v}\right)} = 0 \text{ ,} \\
\pd{(\rho \mbf{v})}{t} + \diverge{\left[ \outerproduct{\rho\mbf{v}}{\mbf{v}} + \left( p_g + \frac{B^2}{2}\right)\underline{\mbf{I}} - \outerproductbf{B}{B}\right]}  = 0 \text{ ,}\\
\pd{E}{t} + \diverge{\left[ \left(E + p_g + \frac{B^2}{2}\right)\mbf{v} - \innerproduct{\mbf{v}}{\outerproductbf{B}{B}}\right]}  = 0 \text{ , and} \\
\pdbf{B}{t} + \diverge{\left[ \outerproductbf{v}{B} - \outerproductbf{B}{v}\right]}  = 0 \text{ ,}
\end{gather*}
where the energy density is defined as
\begin{gather*}
  \label{eqn:energy} E = \frac{p_g}{\gamma - 1} + \frac{\rho v^2}{2} + \frac{B^2}{2}\text{ ,}
\end{gather*}
}


\only<2-2>{
\textbf{Non-strictly} hyperbolic.
\begin{gather*}
\label{eqn:euler_qlin}
\pdbf{U}{t} + \pdbf{F}{x} = 0.
\end{gather*}
Real, but \textbf{not} necessarily distinct eigenvalues:
}
\only<2-2>{
\begin{align*}
v_n &: \text{contact or tangential discontinuity (entropy),} \\
v_n \pm c_s &: \text{slow rarefaction or shock,} \\
v_n \pm c_a &: \text{rotational discontinuity (Alfv{\'e}n), and} \\
v_n \pm c_f &: \text{fast rarefaction or shock,} 
\end{align*}
\begin{gather*}
 c^2_{f,s} = \frac{1}{2}\left[ a^2 + c_a^2 + c_{t}^2 \pm \sqrt{\left(a^2 + c_a^2 + c_{t}^2 \right)^2 - 4a^2c_a^2}\right], 
\end{gather*}
\begin{gather*}
c^2_a = \frac{B_n^2}{\rho} \text{, and } c^2_t = \frac{B_t^2}{\rho}.
\end{gather*}

\bei

\item $\mbf{E} = -\crossproductbf{v}{B}$ at corners.
\item Denote z-component of the emf as $\xi_z$.

\only<1-1>{
Finite area integration of interface $\mbf{B}$
\begin{gather*}
B^{n+1}_{x,i+1/2,j} = B^{n}_{x,i+1/2,j} - \frac{\delta t}{\delta y}\left( \xi_{z,i+1/2,j+1/2} - \xi_{z,i+1/2,j-1/2}\right) \\
B^{n+1}_{y,i,j+1/2} = B^{n}_{y,i,j+1/2} + \frac{\delta t}{\delta x}\left( \xi_{z,i+1/2,j+1/2} - \xi_{z,i-1/2,j-1/2}\right)
\end{gather*}
}
\only<2->{
Due to perfect cancellation, the numerical divergence in the cell remains zero to machine precision.
\begin{gather*}
(\diverge{\mbf{B}})_{i,j} = \frac{1}{\delta x} \left( B_{x,i+1/2,j}  - B_{x,i-1/2,j}\right) + \frac{1}{\delta y} \left( B_{y,i,j+1/2}  - B_{y,i,j-1/2}\right) 
\end{gather*}
}
