\begin{frame}[fragile]
\frametitle{Riemann Problems: Systems of linear equations}
\begin{columns}
\begin{column}{0.65\textwidth}

\only<1-3>{
\bei
\item Hyperbolic $\rightarrow$ real eigenvalues, $\lambda_1, \hdots, \lambda_n$, and linearly independent right eigenvectors, $\mbf{r}^1,\hdots,\mbf{r}^n$.
\item Strictly hyperbolic $\rightarrow$ distinct eigenvalues.
\only<2-3>{
\item Characteristic form, $\mbf{W} = \mbf{l}^i \mbf{U}$, where $\mbf{l}^i$ is the matrix of left eigenvectors.
}
\only<3-3>{
\item $n$ scalar equations.
}
\ebi
}


\only<4->{
\begin{figure}[htbp]
\resizebox{0.9\textwidth}{!}{\tikzsetnextfilename{lin_sys_rstates}\input{fig/lin_sys_rstates.tex}}
\end{figure}
\bei
\item Self-similar solutions for a system of linear waves.
\only<5->{
\item Information is global, state across wave is known.
}
\ebi 
} 

\end{column}
\begin{column}{0.35\textwidth}
\only<1-3>{
\begin{gather*}
\pdbf{U}{t} + \mbf{A}\pdbf{U}{x} = 0,
\end{gather*}
}
\only<2-3>{
\begin{gather*}
\pdbf{W}{t} + \mbf{\Lambda} \pdbf{W}{x} = 0,
\end{gather*}
}
\only<3-3>{
\begin{gather*}
\pd{w_i}{t} + \lambda_i \pd{w_i}{x} = 0,
\end{gather*}
}

\only<4->{
\bei
\item Difference across each wave $\alpha_i = \mbf{l}_i \cdot (\mbf{U}_l - \mbf{U}_r)$.
\ebi 
}
\only<5->{
\bei
\item In terms of conservative variables
$$\mbf{U} = \mbf{U}_l + \sum_{i=1}^m \alpha_i \mbf{r}^i$$ or $$\mbf{U} = \mbf{U}_r - \sum_{i=m+1}^n \alpha_i \mbf{r}^i$$

\ebi
}
\end{column}
\end{columns}
\end{frame}

%----------------------------------------------------------------------------------------
\begin{frame}[fragile]
\frametitle{Compressible hydrodynamics}
\only<1-1>{
\begin{gather*}
\label{eqn:euler1} \pd{\rho}{t} + \diverge{\left(\rho\mbf{v}\right)} = 0 \text{ ,} \\
\label{eqn:euler2} \pd{(\rho \mbf{v})}{t} + \diverge{\left[ \outerproduct{\rho\mbf{v}}{\mbf{v}} + p_g \right]}  = 0 \text{ ,}\\
\label{eqn:euler3} \pd{E}{t} + \diverge{\left[ \left(E + p_g \right)\mbf{v}\right]}  = 0 \text{ ,}
\end{gather*}
where the energy density is defined as
\begin{gather*}
  \label{eqn:energy} E = \frac{p_g}{\gamma - 1} + \frac{\rho v^2}{2}\text{ ,}
\end{gather*}
}


%% \only<2->{
%% 1D quasi-linear form
%% \begin{gather*}
%% \label{eqn:euler_qlin}
%% \pdbf{U}{t} + \mbf{A}(\mbf{U})\pdbf{U}{x} = 0.
%% \end{gather*}
%% }
\only<2-2>{
The Euler equations are strictly hyperbolic.

1D conservative form
\begin{gather*}
\label{eqn:euler_qlin}
\pdbf{U}{t} + \pdbf{F}{x} = 0.
\end{gather*}

Jacobian $\mbf{J}(\mbf{U}) = \partial{\mbf{F}}/\partial{\mbf{U}}$ has three real and distinct eigenvalues:
}
\only<3-3>{
\begin{figure}[htbp]
\resizebox{0.65\textwidth}{!}{\tikzsetnextfilename{euler_rstates}\input{fig/euler_rstates.tex}}
\end{figure}

}
\only<4->{
\begin{tabular}{ccc}
\resizebox{0.33\linewidth}{!}{\tikzsetnextfilename{sod_1}\input{fig/sod_1.tex}} &
\resizebox{0.33\linewidth}{!}{\tikzsetnextfilename{sod_2}\input{fig/sod_2.tex}} &
\resizebox{0.33\linewidth}{!}{\tikzsetnextfilename{sod_3}\input{fig/sod_3.tex}} 
\end{tabular}
}

\only<2-2>{
\begin{align*}
\lambda_3 &= v_n + a : \text{rarefaction or shock} \\
\lambda_2 &= v_n \phantom{+ a}\,\; : \text{contact discontinuity}\\
\lambda_1 &= v_n - a : \text{rarefaction or shock} 
\end{align*}
where $a = \sqrt{\gamma p_g/\rho}$ is the speed of sound.  
}
\only<3-3>{
\begin{align*}
\lambda_3 &= S_l  \phantom{+ a} : \text{rarefaction} \\
\lambda_2 &= v^*_n \phantom{+ a} : \text{contact discontinuity}\\
\lambda_1 &= S_r   \phantom{+ a} : \text{shock} 
\end{align*}
}

\only<4->{
\bei
\item Exact solution to Sod shock tube problem \citep{Sod:1978}.  
\pause
\pause
\pause
\pause
\item Rarefaction head: $x \approx 0.2$.
\pause
\item Rarefaction tail: $x \approx 0.45$.  
\pause
\item Contact discontinuity: $x \approx 0.72$.
\pause
\item Shock: $x \approx 0.95$.
\ebi
}

% Ideal MHD
%----------------------------------------------------------------------------------------
\begin{frame}[fragile]
\frametitle{Ideal magnetohydrodynamics}
\only<1-1>{
\begin{gather*}
\pd{\rho}{t} + \diverge{\left(\rho\mbf{v}\right)} = 0 \text{ ,} \\
\pd{(\rho \mbf{v})}{t} + \diverge{\left[ \outerproduct{\rho\mbf{v}}{\mbf{v}} + \left( p_g + \frac{B^2}{2}\right)\underline{\mbf{I}} - \outerproductbf{B}{B}\right]}  = 0 \text{ ,}\\
\pd{E}{t} + \diverge{\left[ \left(E + p_g + \frac{B^2}{2}\right)\mbf{v} - \innerproduct{\mbf{v}}{\outerproductbf{B}{B}}\right]}  = 0 \text{ , and} \\
\pdbf{B}{t} + \diverge{\left[ \outerproductbf{v}{B} - \outerproductbf{B}{v}\right]}  = 0 \text{ ,}
\end{gather*}
where the energy density is defined as
\begin{gather*}
  \label{eqn:energy} E = \frac{p_g}{\gamma - 1} + \frac{\rho v^2}{2} + \frac{B^2}{2}\text{ ,}
\end{gather*}
}


\only<2-2>{
The ideal MHD equations are non-strictly hyperbolic.
1D conservative form
\begin{gather*}
\label{eqn:euler_qlin}
\pdbf{U}{t} + \pdbf{F}{x} = 0.
\end{gather*}


Jacobian $\mbf{J}(\mbf{U}) = \partial{\mbf{F}}/\partial{\mbf{U}}$ has seven real, but not necessarily distinct eigenvalues:
}
\only<2-2>{
\begin{align*}
v_n &: \text{contact or tangential discontinuity (entropy),} \\
v_n \pm c_s &: \text{slow rarefaction or shock,} \\
v_n \pm c_a &: \text{rotational discontinuity (Alfv{\'e}n), and} \\
v_n \pm c_f &: \text{fast rarefaction or shock,} 
\end{align*}
\begin{gather*}
 c^2_{f,s} = \frac{1}{2}\left[ a^2 + c_a^2 + c_{t}^2 \pm \sqrt{\left(a^2 + c_a^2 + c_{t}^2 \right)^2 - 4a^2c_a^2}\right], 
\end{gather*}
\begin{gather*}
c^2_a = \frac{B_n^2}{\rho} \text{, and } c^2_t = \frac{B_t^2}{\rho}.
\end{gather*} 
 
}

\only<3-3>{
\begin{figure}[htbp]
\resizebox{0.6\textwidth}{!}{\tikzsetnextfilename{mhd_rstates}\input{fig/euler_rstates.tex}}
\end{figure}

}
\only<3-3>{
\begin{align*}
v_n &: \text{contact or tangential discontinuity (entropy),} \\
v_n \pm c_s &: \text{slow rarefaction or shock,} \\
v_n \pm c_a &: \text{rotational discontinuity (Alfv{\'e}n), and} \\
v_n \pm c_f &: \text{fast rarefaction or shock,} 
\end{align*}
}

\end{frame}

\bei
\item Conservative variables are located at the cell centers.
\item Flux is evaluated at cell interface.
\only<1-1>{
\item Flux contribution of each face added to cell. 
\begin{gather*}
\mbf{U}^{n+1}_{\text{cell}} = \mbf{U}^n_{\text{cell}} - \frac{\delta t}{V_{\text{cell}}}\sum_{\text{faces}} \mbf{F}\cdot \mbf{s}
\end{gather*} 
}
\only<2->{
\item In 1D
\begin{gather*}
\mbf{U}^{n+1}_{i} = \mbf{U}^n_{i} - \frac{\delta t}{\delta x}\left(\mbf{F}^n_{i+1/2} - \mbf{F}_{i-1/2}^n \right)
\end{gather*}
}
